367 public static double ParseRatingAPorcentaje(string value)
368 {
369 value = value.Trim();
370
371 // Caso 1: Ya es un porcentaje, como "79%"
372 if (value.EndsWith("%"))
373 {
374 return double.Parse(value.Replace("%", ""), CultureInfo.InvariantCulture);
375 }
376
377 // Caso 2: Es una fracción, como "8.3/10" o "68/100"
378 if (value.Contains("/"))
379 {
380 var parts = value.Split('/');
381 if (parts.Length == 2 && double.TryParse(parts[0], NumberStyles.Any, CultureInfo.InvariantCulture, out double numerator)
382 && double.TryParse(parts[1], NumberStyles.Any, CultureInfo.InvariantCulture, out double denominator))
383 {
384 return (numerator / denominator) * 100;
385 }
386 }else if (value.Contains("."))
387 {
388 double.TryParse(value, NumberStyles.Any, CultureInfo.InvariantCulture, out double entero);
389 return entero * 10;
390 }
391 else
392 {
393 double.TryParse(value, NumberStyles.Any, CultureInfo.InvariantCulture, out double entero);
394 return entero;
395 }
396
397
398 throw new FormatException("Formato de rating no reconocido.");
399 }
